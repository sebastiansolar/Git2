git init - Crear un nuevo repositorio de Git
git status - ver el estado del directorio de trabajo
git add XXXX - Añade un cambio del directorio de trabajo en el area de preparación

git config --global user.email "correo"
git config --global user.name "nombre"


git commit -am "mensaje"  - permite guardar los cambios 
git log - Ver Información sobre los commits


git checkout -- "nombre archivo"   - Me devuelve al ultimo cambio y deshace todos los cambios de mi local

git diff "nombre fichero"     - Nos devuelve el contenido cambiado
